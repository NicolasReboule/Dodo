118template<typename... Features>
119auto VulkanContext::createLogicalDevice(const VulkanContextInfo<Features ...> &ctxInfo) -> std::expected<bool, std::string> {
120 auto graphicsIndex = findGraphicQueueFamily(_physicalDevice);
121 if (!graphicsIndex.has_value())
122 return std::unexpected(graphicsIndex.error());
123
124 auto presentationIndex = findPresentationQueueFamily(_physicalDevice, *graphicsIndex);
125 if (!presentationIndex.has_value())
126 return std::unexpected(presentationIndex.error());
127
128 constexpr float queuePriority = 1.0F;
129 vk::DeviceQueueCreateInfo deviceQueueCreateInfo { .queueFamilyIndex = *graphicsIndex, .queueCount = 1, .pQueuePriorities = &queuePriority };
130
131 const vk::DeviceCreateInfo deviceCreateInfo{
132 .pNext = &getFirstFeature(ctxInfo.featureChain),
133 .queueCreateInfoCount = 1,
134 .pQueueCreateInfos = &deviceQueueCreateInfo,
135 .enabledExtensionCount = static_cast<uint32_t>(ctxInfo.deviceExtensions.size()),
136 .ppEnabledExtensionNames = ctxInfo.deviceExtensions.data()
137 };
138
139 _device = vk::raii::Device(_physicalDevice, deviceCreateInfo);
140 _graphicsQueue = vk::raii::Queue(_device, *graphicsIndex, 0);
141 _presentationQueue = vk::raii::Queue(_device, *presentationIndex, 0);
142 return true;
143}
